## Overview

The `GuildQueuePlayerNode` class manages all playback operations for a guild queue, including playing, pausing, seeking, volume control, and track manipulation. It is accessible via `queue.node`.

## Constructor

```typescript
new GuildQueuePlayerNode<Meta = any>(queue: GuildQueue<Meta>)
```

  The guild queue instance that owns this player node
</ParamField>

## Playback Control Methods

### play()

Plays the given track or the next track in the queue.

```typescript
async play(res?: Track | null, options?: ResourcePlayOptions): Promise<void>
```

<ParamField path="res" type="Track | null" optional>
  The track to play. If not provided, plays the next track from the queue
</ParamField>

<ParamField path="options" type="ResourcePlayOptions" optional>
  Play options
  
  <Expandable title="ResourcePlayOptions properties">
    <ResponseField name="queue" type="boolean">
      If `true`, adds the track to queue instead of playing immediately. Default: `true` if current track exists
    </ResponseField>
    
    <ResponseField name="seek" type="number">
      The position in milliseconds to seek to. Default: `0`
    </ResponseField>
    
    <ResponseField name="transitionMode" type="boolean">
      If `true`, enables transition mode for smooth track changes. Default: `false`
    </ResponseField>
  </Expandable>
</ParamField>

**Example:**

```typescript
// Play next track in queue
await queue.node.play();

// Play a specific track immediately
await queue.node.play(track, { queue: false });

// Play from 30 seconds
await queue.node.play(track, { seek: 30000 });
```

## Progress Methods

### getTimestamp()

Gets the current playback timestamp information.

```typescript
node.getTimestamp(ignoreFilters?: boolean): PlayerTimestamp | null
```

<ParamField path="ignoreFilters" type="boolean" default="false">
  If `true`, returns raw timestamps without filter adjustments
</ParamField>

<ResponseField name="returns" type="PlayerTimestamp | null">
  Timestamp object or `null` if no track is playing
  
  <Expandable title="PlayerTimestamp properties">
    <ResponseField name="current" type="object">
      Current playback position
      
      <Expandable title="properties">
        <ResponseField name="label" type="string">
          Formatted time string (e.g., "1:30")
        </ResponseField>
        
        <ResponseField name="value" type="number">
          Time in milliseconds
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="total" type="object">
      Total track duration
      
      <Expandable title="properties">
        <ResponseField name="label" type="string">
          Formatted time string (e.g., "3:45")
        </ResponseField>
        
        <ResponseField name="value" type="number">
          Duration in milliseconds
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="progress" type="number">
      Progress percentage (0-100)
    </ResponseField>
  </Expandable>
</ResponseField>

**Example:**

```typescript
const timestamp = queue.node.getTimestamp();
if (timestamp) {
  console.log(`${timestamp.current.label} / ${timestamp.total.label}`);
  console.log(`Progress: ${timestamp.progress}%`);
}
```

### createProgressBar()

Creates a visual progress bar for the current track.

```typescript
node.createProgressBar(options?: PlayerProgressbarOptions): string | null
```

<ParamField path="options" type="PlayerProgressbarOptions" optional>
  Progress bar customization options
  
  <Expandable title="PlayerProgressbarOptions properties">
    <ResponseField name="length" type="number" default="15">
      The length of the progress bar in characters
    </ResponseField>
    
    <ResponseField name="timecodes" type="boolean" default="true">
      Whether to show time codes
    </ResponseField>
    
    <ResponseField name="indicator" type="string" default="🔘">
      The character for the current position indicator
    </ResponseField>
    
    <ResponseField name="leftChar" type="string" default="▬">
      The character for elapsed time
    </ResponseField>
    
    <ResponseField name="rightChar" type="string" default="▬">
      The character for remaining time
    </ResponseField>
    
    <ResponseField name="separator" type="string" default="┃">
      The separator between timecode and bar
    </ResponseField>
  </Expandable>
</ParamField>

<ResponseField name="returns" type="string | null">
  The formatted progress bar string, or `null` if no track is playing
</ResponseField>

**Example:**

```typescript
const progressBar = queue.node.createProgressBar();
console.log(progressBar);
// Output: 1:30 ┃ ▬▬▬▬▬🔘▬▬▬▬▬▬▬▬▬ ┃ 3:45

// Custom progress bar
const customBar = queue.node.createProgressBar({
  length: 20,
  indicator: '🎵',
  leftChar: '━',
  rightChar: '─'
});
```
